API Reference

POST /api/generate

Generate FastAPI code from a prompt

Request

{
  "prompt": "Create a user system with JWT authentication",
  "jwt_auth": true,
  "crud_operations": true,
  "user_model": true,
  "database": true,
  "custom_rules": []
}

Response

{
  "project_id": "project_abc123def456",
  "success": true,
  "message": "API generated successfully",
  "files": {
    "models.py": "...",
    "schemas.py": "...",
    "routes.py": "..."
  },
  "validation_summary": { ... }
}

GET /api/projects/{project_id}

Get information about a generated project

Response

{
  "project_id": "project_abc123def456",
  "project_dir": "/path/to/project",
  "files": { ... },
  "metadata": { ... },
  "file_count": 5
}

GET /api/projects/{project_id}/files/{filename}

Get a specific file from a project

Response

{
  "project_id": "project_abc123def456",
  "filename": "models.py",
  "content": "..."
}

PUT /api/projects/{project_id}/files/{filename}

Edit a file in a project

Request

{
  "content": "..."
}

Response

{
  "success": true,
  "message": "Successfully updated models.py",
  "filename": "models.py"
}
